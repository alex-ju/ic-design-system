import { IcDialog, IcButton, IcCheckboxGroup, IcCheckbox } from "@ukic/react";

export const showDialog = (id) => {
  const dialog = document.querySelector(id);
  dialog.showDialog();
};

export const hideDialog = (id) => {
  const dialog = document.querySelector(id);
  dialog.hideDialog();
};

export const confirm = () => {
  alert("Confirmed!");
  hideDialog();
};
